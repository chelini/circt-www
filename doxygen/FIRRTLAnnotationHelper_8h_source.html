38 // The potentially non-local resolved annotation.
39 struct AnnoPathValue {
40  SmallVector<InstanceOp> instances;
41  AnnoTarget ref;
42  unsigned fieldIdx = 0;
43 
44  AnnoPathValue() = default;
45  AnnoPathValue(CircuitOp op) : ref(OpAnnoTarget(op)) {}
46  AnnoPathValue(Operation *op) : ref(OpAnnoTarget(op)) {}
47  AnnoPathValue(const SmallVectorImpl<InstanceOp> &insts, AnnoTarget b,
48  unsigned fieldIdx)
49  : instances(insts.begin(), insts.end()), ref(b), fieldIdx(fieldIdx) {}
50 
51  bool isLocal() const { return instances.empty(); }
52 
53  template <typename... T>
54  bool isOpOfType() const {
55  if (auto opRef = ref.dyn_cast<OpAnnoTarget>())
56  return isa<T...>(opRef.getOp());
57  return false;
58  }
