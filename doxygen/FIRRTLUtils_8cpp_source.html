27 /// Emit a connect between two values.
28 void circt::firrtl::emitConnect(ImplicitLocOpBuilder &builder, Value dst,
29  Value src) {
30  auto dstType = dst.getType().cast<FIRRTLType>();
31  auto srcType = src.getType().cast<FIRRTLType>();
32 
33  // If the types are the exact same we can just connect them.
34  if (dstType == srcType) {
35  // Strict connect does not allow uninferred widths.
36  // TODO: enable strict connect.
37  if (true || dstType.hasUninferredWidth())
38  builder.create<ConnectOp>(dst, src);
39  else
40  builder.create<StrictConnectOp>(dst, src);
41  return;
42  }
43 
44  if (auto dstBundle = dstType.dyn_cast<BundleType>()) {
45  // Connect all the bundle elements pairwise.
46  auto numElements = dstBundle.getNumElements();
47  // Check if we are trying to create an illegal connect - just create the
48  // connect and let the verifier catch it.
49  auto srcBundle = srcType.dyn_cast<BundleType>();
50  if (!srcBundle || numElements != srcBundle.getNumElements()) {
51  builder.create<ConnectOp>(dst, src);
52  return;
53  }
54  for (size_t i = 0; i < numElements; ++i) {
55  auto dstField = builder.create<SubfieldOp>(dst, i);
56  auto srcField = builder.create<SubfieldOp>(src, i);
57  if (dstBundle.getElement(i).isFlip)
58  std::swap(dstField, srcField);
59  emitConnect(builder, dstField, srcField);
60  }
61  return;
62  }
63 
64  if (auto dstVector = dstType.dyn_cast<FVectorType>()) {
65  // Connect all the vector elements pairwise.
66  auto numElements = dstVector.getNumElements();
67  // Check if we are trying to create an illegal connect - just create the
68  // connect and let the verifier catch it.
69  auto srcVector = srcType.dyn_cast<FVectorType>();
70  if (!srcVector || numElements != srcVector.getNumElements()) {
71  builder.create<ConnectOp>(dst, src);
72  return;
73  }
74  for (size_t i = 0; i < numElements; ++i) {
75  auto dstField = builder.create<SubindexOp>(dst, i);
76  auto srcField = builder.create<SubindexOp>(src, i);
77  emitConnect(builder, dstField, srcField);
78  }
79  return;
80  }
81 
82  // Handle ground types with possibly uninferred widths.
83  auto dstWidth = dstType.getBitWidthOrSentinel();
84  auto srcWidth = srcType.getBitWidthOrSentinel();
85  if (dstWidth < 0 || srcWidth < 0) {
86  // If one of these types has an uninferred width, we connect them with a
87  // regular connect operation.
88  builder.create<ConnectOp>(dst, src);
89  return;
90  }
91 
92  // The source must be extended or truncated.
93  if (dstWidth < srcWidth) {
94  // firrtl.tail always returns uint even for sint operands.
95  IntType tmpType = dstType.cast<IntType>();
96  if (tmpType.isSigned())
97  tmpType = UIntType::get(dstType.getContext(), dstWidth);
98  src = builder.create<TailPrimOp>(tmpType, src, srcWidth - dstWidth);
99  // Insert the cast back to signed if needed.
100  if (tmpType != dstType)
101  src = builder.create<AsSIntPrimOp>(dstType, src);
102  } else if (srcWidth < dstWidth) {
103  // Need to extend arg.
104  src = builder.create<PadPrimOp>(src, dstWidth);
105  }
106 
107  // Strict connect requires the types to be completely equal, including
108  // connecting uint<1> to abstract reset types.
109  if (dstType == srcType)
110  builder.create<StrictConnectOp>(dst, src);
111  else
112  builder.create<ConnectOp>(dst, src);
113 }

115 IntegerAttr circt::firrtl::getIntAttr(Type type, const APInt &value) {
116  auto intType = type.cast<IntType>();
117  assert((!intType.hasWidth() ||
118  (unsigned)intType.getWidthOrSentinel() == value.getBitWidth()) &&
119  "value / type width mismatch");
120  auto intSign =
121  intType.isSigned() ? IntegerType::Signed : IntegerType::Unsigned;
122  auto attrType =
123  IntegerType::get(type.getContext(), value.getBitWidth(), intSign);
124  return IntegerAttr::get(attrType, value);
125 }

134 /// Return the value that drives another FIRRTL value within module scope. This
135 /// is parameterized by looking through or not through certain constructs. This
136 /// assumes a single driver and should only be run after `ExpandWhens`.
137 Value circt::firrtl::getModuleScopedDriver(Value val, bool lookThroughWires,
138  bool lookThroughNodes,
139  bool lookThroughCasts) {
140  // Update `val` to the source of the connection driving `thisVal`. This walks
141  // backwards across users to find the first connection and updates `val` to
142  // the source. This assumes that only one connect is driving `thisVal`, i.e.,
143  // this pass runs after `ExpandWhens`.
144  auto updateVal = [&](Value thisVal) {
145  for (auto *user : thisVal.getUsers()) {
146  if (auto connect = dyn_cast<FConnectLike>(user)) {
147  if (connect.dest() != val)
148  continue;
149  val = connect.src();
150  return;
151  }
152  }
153  val = nullptr;
154  return;
155  };
156 
157  while (val) {
158  // The value is a port.
159  if (auto blockArg = val.dyn_cast<BlockArgument>()) {
160  FModuleOp op = cast<FModuleOp>(val.getParentBlock()->getParentOp());
161  auto direction = op.getPortDirection(blockArg.getArgNumber());
162  // Base case: this is one of the module's input ports.
163  if (direction == Direction::In)
164  return blockArg;
165  updateVal(blockArg);
166  continue;
167  }
168 
169  auto *op = val.getDefiningOp();
170 
171  // The value is an instance port.
172  if (auto inst = dyn_cast<InstanceOp>(op)) {
173  auto resultNo = val.cast<OpResult>().getResultNumber();
174  // Base case: this is an instance's output port.
175  if (inst.getPortDirection(resultNo) == Direction::Out)
176  return inst.getResult(resultNo);
177  updateVal(val);
178  continue;
179  }
180 
181  // If told to look through wires, continue from the driver of the wire.
182  if (lookThroughWires && isa<WireOp>(op)) {
183  updateVal(op->getResult(0));
184  continue;
185  }
186 
187  // If told to look through nodes, continue from the node input.
188  if (lookThroughNodes && isa<NodeOp>(op)) {
189  val = cast<NodeOp>(op).input();
190  continue;
191  }
192 
193  if (lookThroughCasts &&
194  isa<AsUIntPrimOp, AsSIntPrimOp, AsClockPrimOp, AsAsyncResetPrimOp>(
195  op)) {
196  val = op->getOperand(0);
197  continue;
198  }
199 
200  // Look through unary ops generated by emitConnect
201  if (isa<PadPrimOp, TailPrimOp>(op)) {
202  val = op->getOperand(0);
203  continue;
204  }
205 
206  // Base case: this is a constant/invalid or primop.
207  //
208  // TODO: If needed, this could be modified to look through unary ops which
209  // have an unambiguous single driver. This should only be added if a need
210  // arises for it.
211  break;
212  };
213  return val;
214 }
