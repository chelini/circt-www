30 /// This just maps a iterator of references to an iterator of addresses.
31 template <typename It>
32 struct AddressIterator
33  : public llvm::mapped_iterator<It, typename It::pointer (*)(
34  typename It::reference)> {
35  // This using statement is to get around a bug in MSVC. Without it, it
36  // tries to look up "It" as a member type of the parent class.

38  /* implicit */ AddressIterator(Iterator iterator)

44 // Escapes any occurance of (regex) characters 'c' in 'str'. If 'noEscape' is

53 // Dumps a range of operations to a string in a format suitable for embedding
54 // inside a .dot edge/node label.
55 template <typename TOpRange>
56 static std::string dotSafeDumpOps(TOpRange ops) {
57  std::string dump;
58  llvm::raw_string_ostream ss(dump);
59  llvm::interleave(
60  ops, ss, [&](mlir::Operation &op) { op.print(ss); }, "\\n");
61 
62  // Ensure that special characters which may be present in the Op dumps are
63  // properly escaped.
64  escape(dump, R"(")");
65  escape(dump, R"(\{)", /*noEscape=*/true);
66  escape(dump, R"(\})", /*noEscape=*/true);
67  return dump;
68 }

251  template <typename Iterator>
252  static std::string getEdgeAttributes(const circt::fsm::FSMStateNode *node,
253  Iterator it, circt::fsm::FSMGraph *) {
254  // Set an edge label that is the dump of the inner contents of the guard of
255  // the transition.
256  circt::fsm::FSMTransitionEdge *edge = *it.getCurrent();
257  circt::fsm::TransitionOp transition = edge->getTransition();
258  if (!transition.hasGuard())
259  return "";
260 
261  std::string attrs = "label=\"";
262  attrs += circt::fsm::detail::dotSafeDumpOps(transition.guard().getOps());
263  attrs += "\"";
264  return attrs;
265  }
